def aviosDF(normal_rate, premium_rate, avios_fv, annual_premium_fee):
    aviosdf = pd.DataFrame(columns = ['AverageMonthlySpend','NormalAvios(1)'
                                     ,'PremiumAvios(1.5)', 'AviosCashDifference'])
    aviosdf['AverageMonthlySpend'] = list(range(500,10000,1))
    aviosdf['NormalAvios(1)'] = aviosdf['AverageMonthlySpend'] * 12 * normal_rate
    aviosdf['PremiumAvios(1.5)'] = aviosdf['AverageMonthlySpend'] * 12 * premium_rate
    aviosdf['AviosCashDifference'] = avios_fv * (aviosdf['PremiumAvios(1.5)'] - aviosdf['NormalAvios(1)'])

    min_spend = aviosdf.loc[(aviosdf['AviosCashDifference'] - 250).abs().argmin()]
    min_spend_y = min_spend.loc['AviosCashDifference']
    min_spend_x = min_spend.loc['AverageMonthlySpend']
    return {'df': aviosdf, 'min_spend' : min_spend, 'min_spend_y' : min_spend_y, 'min_spend_x': min_spend_x}
